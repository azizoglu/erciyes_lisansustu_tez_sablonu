%% İçindekiler Tablosu Çizgileri
\renewcommand*\l@chapter[2]{%
  \ifnum \c@tocdepth >\m@ne
    \addpenalty{-\@highpenalty}%
    \vskip 1.0em \@plus\p@
    \setlength\@tempdima{1.5em}%
    \begingroup
      \parindent \z@ \rightskip \@pnumwidth
      \parfillskip -\@pnumwidth
      \leavevmode \bfseries
      #1 \nobreak
       \if@IcindekilerNoktaliVar
       \leaders\hbox{$\m@th
        \mkern \@dotsep mu\hbox{\bfseries .}\mkern \@dotsep
         mu$}
       \fi
      \hfill\nobreak #2\par
      \penalty\@highpenalty
    \endgroup
  \fi}
  
\def\@tocline#1#2#3#4#5{%
  \vskip 10\p@
  \ifnum #1>\c@tocdepth \else
    \vskip \z@ \@plus.2\p@
    {\leftskip #2\relax \rightskip \@tocrmarg \parfillskip -\rightskip
     \parindent #2\relax\@afterindenttrue
     \interlinepenalty\@M
     \leavevmode
     \@tempdima #3\relax
     \advance\leftskip \@tempdima \null\nobreak\hskip -\leftskip
     {#4}\nobreak
     \if@IcindekilerNoktaliVar
     \leaders\hbox{$\m@th
        \mkern \@dotsep mu\hbox{\bfseries .}\mkern \@dotsep
          mu$}
     \fi
     \hfill
     \nobreak
     \nobreak\protect\bf#5
     \par}%
  \fi}

\def\@toclinenonbold#1#2#3#4#5{%
  \vskip 10\p@
  \ifnum #1>\c@tocdepth \else
    \vskip \z@ \@plus.2\p@
    {\leftskip #2\relax \rightskip \@tocrmarg \parfillskip -\rightskip
     \parindent #2\relax\@afterindenttrue
     \interlinepenalty\@M
     \leavevmode
     \@tempdima #3\relax
     \advance\leftskip \@tempdima \null\nobreak\hskip -\leftskip
     {#4}\nobreak
     \if@IcindekilerNoktaliVar
     \leaders\hbox{$\m@th
        \mkern \@dotsep mu\hbox{.}\mkern \@dotsep
          mu$}
     \fi
     \hfill
     \nobreak
     \nobreak#5
     \par}%
  \fi}

  \thesection.\ & \parbox[t]{\textwidth - \widthof{\thesection.\ } - 1em}{#2}
  \end{tabular}
  \endgroup\goodbreak
  \vspace*{-2mm}\goodbreak
  \setlength{\baslikGenisligi}{\widthof{#2}}
  \setlength{\metinGenisligi}{\textwidth - \widthof{\thesection.\ }}
  \ifthenelse{\baslikGenisligi > \metinGenisligi}{\vspace{1em}}{}
  \addcontentsline{toc}{section}{\protect\bf\protect\numberline{\thesection.}\protect\bf #1}}

%% Chapter (Bölüm) Tanımları
\renewcommand\chapter{\if@openright\cleardoublepage\else\clearpage\fi
                      \global\@topnum\z@
                      \@afterindentfalse
                      \secdef\@chapter\@schapter}

\def\@chapter[#1]#2{\ifnum \c@secnumdepth >\m@ne
                         \refstepcounter{chapter}%
            \if@EklerVar
            \else
                \if@BolumBoldVar
                    \addcontentsline{toc}{bolumbaslik}{\thechapter. \bolumToC \protect\\[5pt] \MakeTextUppercase{#1}}
                \else
                    \addcontentsline{toc}{bolumbasliknormal}{\thechapter. \bolumToC \protect\\[5pt] \MakeTextUppercase{#1}}
                \fi
            \fi
                    \else
                      \relax
                    \fi
                    \chaptermark{#1}%
                    \if@twocolumn
                      \@topnewpage[\@makechapterhead{#2}]%
                    \else
                      \@makechapterhead{#2}%
                      \@afterheading
        \fi
                    }

\def\@makechapterhead#1{
    \thispagestyle{empty}%
    \vspace*{\bolumUstBosluk}%
    \vspace*{-7mm}
    {\parindent \z@ \centering 
    \normalfont
    \ifnum \c@secnumdepth >\m@ne
        \if@EklerVar
            \th@BaslikFont\th@BolumBoyut \centering \MakeTextUppercase{\thechapter}
        \else
            \th@BaslikFont\th@BolumBoyut \MakeTextUppercase{\thechapter. \@chapapp}
        \fi
        \par\nobreak
        \vskip 10\p@
    \fi
    \interlinepenalty\@M
    \th@BaslikFont\th@BolumBoyut \MakeTextUppercase{#1}\par\nobreak
    \vskip 5\p@
  }}

\def\@schapter#1{
    \@makeschapterhead{#1}
    \@afterheading
    }
    
\def\@makeschapterhead#1{%
  \vspace*{\bolumUstBosluk}%
  {\parindent \z@ \centering
    \normalfont
    \interlinepenalty\@M
    \th@BaslikFont\th@BolumBoyut \MakeTextUppercase{#1}\par\nobreak
    \vskip 20\p@}}
